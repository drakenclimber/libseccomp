65 static int _rc_filter(int err)
66 {
67  /* pass through success values */
68  if (err >= 0)
69  return err;
70 
71  /* filter the error codes */
72  switch (err) {
73  case -EACCES:
74  /* NOTE: operation is not permitted by libseccomp */
75  case -ECANCELED:
76  /* NOTE: kernel level error that is beyond the control of
77  * libseccomp */
78  case -EDOM:
79  /* NOTE: failure due to arch/ABI */
80  case -EEXIST:
81  /* NOTE: operation failed due to existing rule or filter */
82  case -EINVAL:
83  /* NOTE: invalid input to the libseccomp API */
84  case -ENOENT:
85  /* NOTE: no matching entry found */
86  case -ENOMEM:
87  /* NOTE: unable to allocate enough memory to perform the
88  * requested operation */
89  case -EOPNOTSUPP:
90  /* NOTE: operation is not supported */
91  case -ERANGE:
92  /* NOTE: provided buffer is too small */
93  case -ESRCH:
94  /* NOTE: operation failed due to multi-threading */
95  return err;
96  default:
97  /* NOTE: this is the default "internal libseccomp error"
98  * error code, it is our catch-all */
99  return -EFAULT;
100  }
101 }

165 static unsigned int _seccomp_api_update(void)
166 {
167  unsigned int level = 1;
168 
169  /* if seccomp_api_level > 0 then it's already been set, we're done */
170  if (seccomp_api_level >= 1)
171  return seccomp_api_level;
172 
173  /* NOTE: level 1 is the base level, start checking at 2 */
174 
175  if (sys_chk_seccomp_syscall() &&
176  sys_chk_seccomp_flag(SECCOMP_FILTER_FLAG_TSYNC) == 1)
177  level = 2;
178 
179  if (level == 2 &&
180  sys_chk_seccomp_flag(SECCOMP_FILTER_FLAG_LOG) == 1 &&
181  sys_chk_seccomp_action(SCMP_ACT_LOG) == 1 &&
182  sys_chk_seccomp_action(SCMP_ACT_KILL_PROCESS) == 1)
183  level = 3;
184 
185  if (level == 3 &&
186  sys_chk_seccomp_flag(SECCOMP_FILTER_FLAG_SPEC_ALLOW) == 1)
187  level = 4;
188 
189  if (level == 4 &&
190  sys_chk_seccomp_flag(SECCOMP_FILTER_FLAG_NEW_LISTENER) == 1 &&
191  sys_chk_seccomp_action(SCMP_ACT_NOTIFY) == 1)
192  level = 5;
193 
194  if (level == 5 &&
195  sys_chk_seccomp_flag(SECCOMP_FILTER_FLAG_TSYNC_ESRCH) == 1)
196  level = 6;
197 
198  /* update the stored api level and return */
199  seccomp_api_level = level;
200  return seccomp_api_level;
201 }
