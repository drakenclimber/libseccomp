112 int sys_chk_seccomp_syscall(void)
113 {
114  int rc;
115  int nr_seccomp;
116 
117  /* NOTE: it is reasonably safe to assume that we should be able to call
118  * seccomp() when the caller first starts, but we can't rely on
119  * it later so we need to cache our findings for use later */
120  if (state.sup_syscall >= 0)
121  return state.sup_syscall;
122 
123 #if SYSCALL_ALLOWLIST_ENABLE
124  /* architecture allowlist */
125  switch (arch_def_native->token) {
126  case SCMP_ARCH_X86_64:
127  case SCMP_ARCH_ARM:
128  case SCMP_ARCH_AARCH64:
129  case SCMP_ARCH_PPC64:
130  case SCMP_ARCH_PPC64LE:
131  case SCMP_ARCH_S390:
132  case SCMP_ARCH_S390X:
133  case SCMP_ARCH_RISCV64:
134  break;
135  default:
136  goto unsupported;
137  }
138 #endif
139 
140  nr_seccomp = arch_syscall_resolve_name(arch_def_native, "seccomp");
141  if (nr_seccomp < 0)
142  goto unsupported;
143 
144  /* this is an invalid call because the second argument is non-zero, but
145  * depending on the errno value of ENOSYS or EINVAL we can guess if the
146  * seccomp() syscall is supported or not */
147  rc = syscall(nr_seccomp, SECCOMP_SET_MODE_STRICT, 1, NULL);
148  if (rc < 0 && errno == EINVAL)
149  goto supported;
150 
151 unsupported:
152  state.sup_syscall = 0;
153  return 0;
154 supported:
155  state.nr_seccomp = nr_seccomp;
156  state.sup_syscall = 1;
157  return 1;
158 }

181 int sys_chk_seccomp_action(uint32_t action)
182 {
183  if (action == SCMP_ACT_KILL_PROCESS) {
184  if (state.sup_kill_process < 0) {
185  if (sys_chk_seccomp_syscall() == 1 &&
186  syscall(state.nr_seccomp,
187  SECCOMP_GET_ACTION_AVAIL, 0, &action) == 0)
188  state.sup_kill_process = 1;
189  else
190  state.sup_kill_process = 0;
191  }
192 
193  return state.sup_kill_process;
194  } else if (action == SCMP_ACT_KILL_THREAD) {
195  return 1;
196  } else if (action == SCMP_ACT_TRAP) {
197  return 1;
198  } else if ((action == SCMP_ACT_ERRNO(action & 0x0000ffff)) &&
199  ((action & 0x0000ffff) < MAX_ERRNO)) {
200  return 1;
201  } else if (action == SCMP_ACT_TRACE(action & 0x0000ffff)) {
202  return 1;
203  } else if (action == SCMP_ACT_LOG) {
204  if (state.sup_action_log < 0) {
205  if (sys_chk_seccomp_syscall() == 1 &&
206  syscall(state.nr_seccomp,
207  SECCOMP_GET_ACTION_AVAIL, 0, &action) == 0)
208  state.sup_action_log = 1;
209  else
210  state.sup_action_log = 0;
211  }
212 
213  return state.sup_action_log;
214  } else if (action == SCMP_ACT_ALLOW) {
215  return 1;
216  } else if (action == SCMP_ACT_NOTIFY) {
217  if (state.sup_user_notif < 0) {
218  struct seccomp_notif_sizes sizes;
219  if (sys_chk_seccomp_syscall() == 1 &&
220  syscall(state.nr_seccomp,
221  SECCOMP_GET_NOTIF_SIZES, 0, &sizes) == 0)
222  state.sup_user_notif = 1;
223  else
224  state.sup_user_notif = 0;
225  }
226 
227  return state.sup_user_notif;
228  }
229 
230  return 0;
231 }

264 static int _sys_chk_flag_kernel(int flag)
265 {
266  /* this is an invalid seccomp(2) call because the last argument
267  * is NULL, but depending on the errno value of EFAULT we can
268  * guess if the filter flag is supported or not */
269  if (sys_chk_seccomp_syscall() == 1 &&

272  errno == EFAULT)
273  return 1;

287 int sys_chk_seccomp_flag(int flag)
288 {
289  switch (flag) {
290  case SECCOMP_FILTER_FLAG_TSYNC:
291  if (state.sup_flag_tsync < 0)
292  state.sup_flag_tsync = _sys_chk_flag_kernel(flag);
293  return state.sup_flag_tsync;
294  case SECCOMP_FILTER_FLAG_LOG:
295  if (state.sup_flag_log < 0)
296  state.sup_flag_log = _sys_chk_flag_kernel(flag);
297  return state.sup_flag_log;
298  case SECCOMP_FILTER_FLAG_SPEC_ALLOW:
299  if (state.sup_flag_spec_allow < 0)
300  state.sup_flag_spec_allow = _sys_chk_flag_kernel(flag);
301  return state.sup_flag_spec_allow;
302  case SECCOMP_FILTER_FLAG_NEW_LISTENER:
303  if (state.sup_flag_new_listener < 0)
304  state.sup_flag_new_listener = _sys_chk_flag_kernel(flag);
305  return state.sup_flag_new_listener;
306  case SECCOMP_FILTER_FLAG_TSYNC_ESRCH:
307  if (state.sup_flag_tsync_esrch < 0)
308  state.sup_flag_tsync_esrch = _sys_chk_flag_kernel(flag);
309  return state.sup_flag_tsync_esrch;
310  }
311 
312  return -EOPNOTSUPP;
313 }

356 int sys_filter_load(struct db_filter_col *col, bool rawrc)
357 {
358  int rc;
359  bool tsync_notify;
360  bool listener_req;
361  struct bpf_program *prgm = NULL;
362 
363  rc = gen_bpf_generate(col, &prgm);
364  if (rc < 0)
365  return rc;
366 
367  /* attempt to set NO_NEW_PRIVS */
368  if (col->attr.nnp_enable) {
369  rc = prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0);
370  if (rc < 0)
371  goto filter_load_out;
372  }
373 
374  tsync_notify = state.sup_flag_tsync_esrch > 0 && state.notify_fd == -1;
375  listener_req = state.sup_user_notif > 0 && \
376  col->notify_used && state.notify_fd == -1;
377 
378  /* load the filter into the kernel */
379  if (sys_chk_seccomp_syscall() == 1) {
380  int flgs = 0;
381  if (tsync_notify) {
382  if (col->attr.tsync_enable)
383  flgs |= SECCOMP_FILTER_FLAG_TSYNC | \
384  SECCOMP_FILTER_FLAG_TSYNC_ESRCH;
385  if (listener_req)
386  flgs |= SECCOMP_FILTER_FLAG_NEW_LISTENER;
387  } else if (col->attr.tsync_enable) {
388  if (listener_req) {
389  /* NOTE: we _should_ catch this in db.c */
390  rc = -EFAULT;
391  goto filter_load_out;
392  }
393  flgs |= SECCOMP_FILTER_FLAG_TSYNC;
394  } else if (listener_req)
395  flgs |= SECCOMP_FILTER_FLAG_NEW_LISTENER;
396  if (col->attr.log_enable)
397  flgs |= SECCOMP_FILTER_FLAG_LOG;
398  if (col->attr.spec_allow)
399  flgs |= SECCOMP_FILTER_FLAG_SPEC_ALLOW;
400  rc = syscall(state.nr_seccomp,
401  SECCOMP_SET_MODE_FILTER, flgs, prgm);
402  if (tsync_notify && rc > 0) {
403  /* return 0 on NEW_LISTENER success, but save the fd */
404  state.notify_fd = rc;
405  rc = 0;
406  } else if (rc > 0 && col->attr.tsync_enable) {
407  /* always return -ESRCH if we fail to sync threads */
408  errno = ESRCH;
409  rc = -errno;
410  } else if (rc > 0 && state.sup_user_notif > 0) {
411  /* return 0 on NEW_LISTENER success, but save the fd */
412  state.notify_fd = rc;
413  rc = 0;
414  }
415  } else
416  rc = prctl(PR_SET_SECCOMP, SECCOMP_MODE_FILTER, prgm);
417 
418 filter_load_out:
419  /* cleanup and return */
420  gen_bpf_release(prgm);
421  if (rc == -ESRCH)
422  return -ESRCH;
423  if (rc < 0)
424  return (rawrc ? -errno : -ECANCELED);
425  return rc;
426 }

451 int sys_notify_alloc(struct seccomp_notif **req,
452  struct seccomp_notif_resp **resp)
453 {
454  int rc;
455  static struct seccomp_notif_sizes sizes = { 0, 0, 0 };
456 
457  if (state.sup_syscall <= 0)
458  return -EOPNOTSUPP;
459 
460  if (sizes.seccomp_notif == 0 && sizes.seccomp_notif_resp == 0) {
461  rc = syscall(__NR_seccomp, SECCOMP_GET_NOTIF_SIZES, 0, &sizes);
462  if (rc < 0)
463  return -ECANCELED;
464  }
465  if (sizes.seccomp_notif == 0 || sizes.seccomp_notif_resp == 0)
466  return -EFAULT;
467 
468  if (req) {
469  *req = zmalloc(sizes.seccomp_notif);
470  if (!*req)
471  return -ENOMEM;
472  }
473 
474  if (resp) {
475  *resp = zmalloc(sizes.seccomp_notif_resp);
476  if (!*resp) {
477  if (req)
478  free(*req);
479  return -ENOMEM;
480  }
481  }
482 
483  return 0;
484 }

548  * kernel commit 47e33c05f9f0 ("seccomp: Fix ioctl number for
549  * SECCOMP_IOCTL_NOTIF_ID_VALID"). Try the old value. */
550  rc = ioctl(fd, SECCOMP_IOCTL_NOTIF_ID_VALID_WRONG_DIR, &id);
551  if (rc < 0)
552  return -ENOENT;
