113 static void end_action(uint32_t action, unsigned int line)
114 {
115  uint32_t act = action & SECCOMP_RET_ACTION_FULL;
116  uint32_t data = action & SECCOMP_RET_DATA;
117 
118  switch (act) {
119  case SECCOMP_RET_KILL_PROCESS:
120  fprintf(stdout, "KILL_PROCESS\n");
121  break;
122  case SECCOMP_RET_KILL_THREAD:
123  fprintf(stdout, "KILL\n");
124  break;
125  case SECCOMP_RET_TRAP:
126  fprintf(stdout, "TRAP\n");
127  break;
128  case SECCOMP_RET_ERRNO:
129  fprintf(stdout, "ERRNO(%u)\n", data);
130  break;
131  case SECCOMP_RET_TRACE:
132  fprintf(stdout, "TRACE(%u)\n", data);
133  break;
134  case SECCOMP_RET_LOG:
135  fprintf(stdout, "LOG\n");
136  break;
137  case SECCOMP_RET_ALLOW:
138  fprintf(stdout, "ALLOW\n");
139  break;
140  default:
141  exit_error(EDOM, line);
142  }

155 static void bpf_execute(const struct bpf_program *prg,
156  const struct seccomp_data *sys_data)
157 {
158  unsigned int ip, ip_c;
159  struct sim_state state;
160  bpf_instr_raw *bpf;
161  unsigned char *sys_data_b = (unsigned char *)sys_data;
162  uint16_t code;
163  uint8_t jt;
164  uint8_t jf;
165  uint32_t k;
166 
167  /* initialize the machine state */
168  ip_c = 0;
169  ip = 0;
170  memset(&state, 0, sizeof(state));
171 
172  while (ip < prg->i_cnt) {
173  /* get the instruction and bump the ip */
174  ip_c = ip;
175  bpf = &prg->i[ip++];
176 
177  code = ttoh16(arch, bpf->code);
178  jt = bpf->jt;
179  jf = bpf->jf;
180  k = ttoh32(arch, bpf->k);
181 
182  switch (code) {
183  case BPF_LD+BPF_W+BPF_ABS:
184  if (k < BPF_SYSCALL_MAX) {
185  uint32_t val = *((uint32_t *)&sys_data_b[k]);
186  state.acc = ttoh32(arch, val);
187  } else
188  exit_error(ERANGE, ip_c);
189  break;
190  case BPF_ALU+BPF_OR+BPF_K:
191  state.acc |= k;
192  break;
193  case BPF_ALU+BPF_AND+BPF_K:
194  state.acc &= k;
195  break;
196  case BPF_JMP+BPF_JA:
197  ip += k;
198  break;
199  case BPF_JMP+BPF_JEQ+BPF_K:
200  if (state.acc == k)
201  ip += jt;
202  else
203  ip += jf;
204  break;
205  case BPF_JMP+BPF_JGT+BPF_K:
206  if (state.acc > k)
207  ip += jt;
208  else
209  ip += jf;
210  break;
211  case BPF_JMP+BPF_JGE+BPF_K:
212  if (state.acc >= k)
213  ip += jt;
214  else
215  ip += jf;
216  break;
217  case BPF_RET+BPF_K:
218  end_action(k, ip_c);
219  break;
220  default:
221  /* since we don't support the full bpf language just
222  * yet, this could be either a fault or an error, we'll
223  * treat it as a fault until we provide full support */
224  exit_fault(EOPNOTSUPP);
225  }
226  }
227 
228  /* if we've reached here there is a problem with the program */
229  exit_error(ERANGE, ip_c);
230 }
