179 static void _gen_pfc_chain(const struct arch_def *arch,
180  const struct db_arg_chain_tree *node,
181  unsigned int lvl, FILE *fds)
182 {
183  const struct db_arg_chain_tree *c_iter;
184 
185  /* get to the start */
186  c_iter = node;
187  while (c_iter->lvl_prv != NULL)
188  c_iter = c_iter->lvl_prv;
189 
190  while (c_iter != NULL) {
191  /* comparison operation */
192  _indent(fds, lvl);
193  fprintf(fds, "if (");
194  _pfc_arg(fds, arch, c_iter);
195  switch (c_iter->op) {
196  case SCMP_CMP_EQ:
197  fprintf(fds, " == ");
198  break;
199  case SCMP_CMP_GE:
200  fprintf(fds, " >= ");
201  break;
202  case SCMP_CMP_GT:
203  fprintf(fds, " > ");
204  break;
205  case SCMP_CMP_MASKED_EQ:
206  fprintf(fds, " & 0x%.8x == ", c_iter->mask);
207  break;
208  case SCMP_CMP_NE:
209  case SCMP_CMP_LT:
210  case SCMP_CMP_LE:
211  default:
212  fprintf(fds, " ??? ");
213  }
214  fprintf(fds, "%u)\n", c_iter->datum);
215 
216  /* true result */
217  if (c_iter->act_t_flg) {
218  _indent(fds, lvl + 1);
219  _pfc_action(fds, c_iter->act_t);
220  } else if (c_iter->nxt_t != NULL)
221  _gen_pfc_chain(arch, c_iter->nxt_t, lvl + 1, fds);
222 
223  /* false result */
224  if (c_iter->act_f_flg) {
225  _indent(fds, lvl);
226  fprintf(fds, "else\n");
227  _indent(fds, lvl + 1);
228  _pfc_action(fds, c_iter->act_f);
229  } else if (c_iter->nxt_f != NULL) {
230  _indent(fds, lvl);
231  fprintf(fds, "else\n");
232  _gen_pfc_chain(arch, c_iter->nxt_f, lvl + 1, fds);
233  }
234 
235  c_iter = c_iter->lvl_nxt;
236  }
237 }

303  struct pfc_sys_list **p_head)
304 {
305  struct pfc_sys_list *p_iter = NULL, *p_new, *p_prev;
306  struct db_sys_list *s_iter;
307 
308  db_list_foreach(s_iter, syscalls) {
309  p_new = zmalloc(sizeof(*p_new));
310  if (p_new == NULL) {
311  return -ENOMEM;
312  }
313  p_new->sys = s_iter;
314 
315  p_prev = NULL;
316  p_iter = *p_head;
317  while (p_iter != NULL &&
318  s_iter->num < p_iter->sys->num) {
319  p_prev = p_iter;
320  p_iter = p_iter->next;
321  }
322  if (*p_head == NULL)
323  *p_head = p_new;
324  else if (p_prev == NULL) {
325  p_new->next = *p_head;
326  *p_head = p_new;
327  } else {
328  p_new->next = p_iter;
329  p_prev->next = p_new;
330  }
331  }
332 
333  return 0;
334 }

392 static int _gen_pfc_arch(const struct db_filter_col *col,
393  const struct db_filter *db, FILE *fds,
394  uint32_t optimize)
395 {
396  int rc = 0, i = 0, lookahead_num;
397  unsigned int syscall_cnt = 0, bintree_levels, level, indent = 1;
398  struct pfc_sys_list *p_iter = NULL, *p_head = NULL;
399 
400  /* sort the syscall list */
401  rc = _sys_sort(db->syscalls, &p_head, optimize);
402  if (rc < 0)
403  goto arch_return;
404 
405  bintree_levels = _get_bintree_levels(db->syscall_cnt, optimize);
406 
407  fprintf(fds, "# filter for arch %s (%u)\n",
408  _pfc_arch(db->arch), db->arch->token_bpf);
409  fprintf(fds, "if ($arch == %u)\n", db->arch->token_bpf);
410  p_iter = p_head;
411  while (p_iter != NULL) {
412  if (!p_iter->sys->valid) {
413  p_iter = p_iter->next;
414  continue;
415  }
416 
417  for (i = bintree_levels - 1; i > 0; i--) {
418  level = SYSCALLS_PER_NODE << i;
419 
420  if (syscall_cnt == 0 || (syscall_cnt % level) == 0) {
421  rc = _get_bintree_syscall_num(p_iter, level / 2,
422  &lookahead_num);
423  if (rc < 0)
424  /* We have reached the end of the bintree.
425  * There aren't enough syscalls to construct
426  * any more if-elses.
427  */
428  continue;
429  _indent(fds, indent);
430  fprintf(fds, "if ($syscall > %u)\n", lookahead_num);
431  indent++;
432  } else if ((syscall_cnt % (level / 2)) == 0) {
433  lookahead_num = p_iter->sys->num;
434  _indent(fds, indent - 1);
435  fprintf(fds, "else # ($syscall <= %u)\n",
436  p_iter->sys->num);
437  }
438 
439  }
440 
441  _gen_pfc_syscall(db->arch, p_iter->sys, fds, indent);
442  syscall_cnt++;
443  p_iter = p_iter->next;
444 
445  /* undo the indentations as the else statements complete */
446  for (i = 0; i < bintree_levels; i++) {
447  if (syscall_cnt % ((SYSCALLS_PER_NODE * 2) << i) == 0)
448  indent--;
449  }
450  }
451  _indent(fds, 1);
452  fprintf(fds, "# default action\n");
453  _indent(fds, 1);
454  _pfc_action(fds, col->attr.act_default);
455 
456 arch_return:
457  while (p_head != NULL) {
458  p_iter = p_head;
459  p_head = p_head->next;
460  free(p_iter);
461  }
462  return rc;
463 }
474 int gen_pfc_generate(const struct db_filter_col *col, int fd)
475 {
476  int newfd;
477  unsigned int iter;

479 
480  newfd = dup(fd);
481  if (newfd < 0)
482  return -errno;
483  fds = fdopen(newfd, "a");
484  if (fds == NULL) {
485  close(newfd);
486  return -errno;
487  }
488 
489  /* generate the pfc */
490  fprintf(fds, "#\n");
491  fprintf(fds, "# pseudo filter code start\n");
492  fprintf(fds, "#\n");
493

495  _gen_pfc_arch(col, col->filters[iter], fds,
496  col->attr.optimize);
497 
498  fprintf(fds, "# invalid architecture action\n");
499  _pfc_action(fds, col->attr.act_badarch);
500  fprintf(fds, "#\n");
501  fprintf(fds, "# pseudo filter code end\n");
502  fprintf(fds, "#\n");
503 
504  fflush(fds);
505  fclose(fds);
506 
507  return 0;
