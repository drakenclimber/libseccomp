52 /*
53  * All BPF programs must return a 32-bit value.
54  * The bottom 16-bits are for optional return data.
55  * The upper 16-bits are ordered from least permissive values to most.
56  *
57  * The ordering ensures that a min_t() over composed return values always
58  * selects the least permissive choice.
59  */
60 #define SECCOMP_RET_KILL_PROCESS 0x80000000U /* kill the process immediately */
61 #define SECCOMP_RET_KILL_THREAD 0x00000000U /* kill the thread immediately */
62 #define SECCOMP_RET_KILL SECCOMP_RET_KILL_THREAD /* default to killing the thread */
63 #define SECCOMP_RET_TRAP 0x00030000U /* disallow and force a SIGSYS */
64 #define SECCOMP_RET_ERRNO 0x00050000U /* returns an errno */
65 #define SECCOMP_RET_USER_NOTIF 0x7fc00000U /* notifies userspace */
66 #define SECCOMP_RET_TRACE 0x7ff00000U /* pass to a tracer or disallow */
67 #define SECCOMP_RET_ALLOW 0x7fff0000U /* allow */
