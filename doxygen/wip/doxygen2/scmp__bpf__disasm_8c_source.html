215 static void bpf_decode_args(const bpf_instr_raw *bpf, unsigned int line)
216 {
217  switch (BPF_CLASS(bpf->code)) {
218  case BPF_LD:
219  case BPF_LDX:
220  switch (BPF_MODE(bpf->code)) {
221  case BPF_ABS:
222  printf("$data[%u]", bpf->k);
223  break;
224  case BPF_MEM:
225  printf("$temp[%u]", bpf->k);
226  break;
227  case BPF_IMM:
228  printf("%u", bpf->k);
229  break;
230  case BPF_IND:
231  printf("$data[X + %u]", bpf->k);
232  break;
233  case BPF_LEN:
234  printf("len($data)");
235  break;
236  case BPF_MSH:
237  printf("4 * $data[%u] & 0x0f", bpf->k);
238  break;
239  }
240  break;
241  case BPF_ST:
242  case BPF_STX:
243  printf("$temp[%u]", bpf->k);
244  break;
245  case BPF_ALU:
246  if (BPF_SRC(bpf->code) == BPF_K) {
247  switch (BPF_OP(bpf->code)) {
248  case BPF_OR:
249  case BPF_AND:
250  printf("0x%.8x", bpf->k);
251  break;
252  default:
253  printf("%u", bpf->k);
254  }
255  } else
256  printf("%u", bpf->k);
257  break;
258  case BPF_JMP:
259  if (BPF_OP(bpf->code) == BPF_JA) {
260  printf("%.4u", (line + 1) + bpf->k);
261  } else {
262  printf("%-4u true:%.4u false:%.4u",
263  bpf->k,
264  (line + 1) + bpf->jt,
265  (line + 1) + bpf->jf);
266  }
267  break;
268  case BPF_RET:
269  if (BPF_RVAL(bpf->code) == BPF_A) {
270  /* XXX - accumulator? */
271  printf("$acc");
272  } else if (BPF_SRC(bpf->code) == BPF_K) {
273  bpf_decode_action(bpf->k);
274  } else if (BPF_SRC(bpf->code) == BPF_X) {
275  /* XXX - any idea? */
276  printf("???");
277  }
278  break;
279  case BPF_MISC:
280  break;
281  default:
282  printf("???");
283  }

294 static int bpf_decode(FILE *file)
295 {
296  unsigned int line = 0;
297  bpf_instr_raw bpf;
298 
299  /* header */
300  printf(" line OP JT JF K\n");
301  printf("=================================\n");
302 
303  while (fread(&bpf, sizeof(bpf), 1, file)) {
304  /* convert the bpf statement */
305  bpf.code = ttoh16(arch, bpf.code);
306  bpf.k = ttoh32(arch, bpf.k);
307 
308  /* display a hex dump */
309  printf(" %.4u: 0x%.2x 0x%.2x 0x%.2x 0x%.8x",
310  line, bpf.code, bpf.jt, bpf.jf, bpf.k);
311 
312  /* display the assembler statements */
313  printf(" ");
314  printf(_OP_FMT, bpf_decode_op(&bpf));
315  printf(" ");
316  bpf_decode_args(&bpf, line);
317  printf("\n");
318 
319  line++;
320  }
321 
322  if (ferror(file))
323  return errno;
324  return 0;
325 }

429 static int bpf_dot_decode(FILE *file)
430 {
431  unsigned int line = 0;
432  bpf_instr_raw bpf;
433  int prev_class = 0;
434 
435  /* header */
436  printf("digraph {\n");
437  printf("\tstart[shape=\"box\", style=rounded];\n");
438 
439  while (fread(&bpf, sizeof(bpf), 1, file)) {
440  /* convert the bpf statement */
441  bpf.code = ttoh16(arch, bpf.code);
442  bpf.k = ttoh32(arch, bpf.k);
443 
444  /* display the statement */
445  bpf_dot_decode_args(&bpf, line);
446 
447  /* if previous line wasn't RET/JMP, link it to this line */
448  if (line == 0)
449  printf("\tstart -> line%d\n", line);
450  else if ((prev_class != BPF_JMP) && (prev_class != BPF_RET))
451  printf("\tline%d -> line%d\n", line - 1, line);
452  prev_class = BPF_CLASS(bpf.code);
453 
454  line++;
455  }
456  printf("}\n");
457 
458  if (ferror(file))
459  return errno;
460  return 0;
461 }
