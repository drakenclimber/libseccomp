246 int arch_arg_offset_lo(const struct arch_def *arch, unsigned int arg)
247 {

355 int arch_syscall_translate(const struct arch_def *arch, int *syscall)
356 {
357  int sc_num;
358  const char *sc_name;
359 
360  /* special handling for syscall -1 */
361  if (*syscall == -1)
362  return 0;
363 
364  if (arch->token != arch_def_native->token) {
365  sc_name = arch_syscall_resolve_num(arch_def_native, *syscall);
366  if (sc_name == NULL)
367  return -EFAULT;
368 
369  sc_num = arch_syscall_resolve_name(arch, sc_name);
370  if (sc_num == __NR_SCMP_ERROR)
371  return -EFAULT;
372 
373  *syscall = sc_num;
374  }
375 
376  return 0;
377 }

390 int arch_syscall_rewrite(const struct arch_def *arch, int *syscall)
391 {
392  int sys = *syscall;
393 
394  if (sys >= -1) {
395  /* we shouldn't be here - no rewrite needed */
396  return 0;
397  } else if (sys > -100) {
398  /* -2 to -99 are reserved values */
399  return -EINVAL;
400  } else if (sys > -10000) {
401  /* rewritable syscalls */
402  if (arch->syscall_rewrite)
403  (*arch->syscall_rewrite)(arch, syscall);
404  }
405 
406  /* syscalls not defined on this architecture */
407  if ((*syscall) < 0)
408  return -EDOM;
409  return 0;
410 }

430 int arch_filter_rule_add(struct db_filter *db,
431  const struct db_api_rule_list *rule)
432 {
433  int rc = 0;
434  int syscall;
435  struct db_api_rule_list *rule_dup = NULL;
436 
437  /* create our own rule that we can munge */
438  rule_dup = db_rule_dup(rule);
439  if (rule_dup == NULL)
440  return -ENOMEM;
441 
442  /* translate the syscall */
443  rc = arch_syscall_translate(db->arch, &rule_dup->syscall);
444  if (rc < 0)
445  goto rule_add_return;
446  syscall = rule_dup->syscall;
447 
448  /* add the new rule to the existing filter */
449  if (syscall == -1 || db->arch->rule_add == NULL) {
450  /* syscalls < -1 require a db->arch->rule_add() function */
451  if (syscall < -1 && rule_dup->strict) {
452  rc = -EDOM;
453  goto rule_add_return;
454  }
455  rc = db_rule_add(db, rule_dup);
456  } else
457  rc = (db->arch->rule_add)(db, rule_dup);
458 
459 rule_add_return:
460  /* NOTE: another reminder that we don't do any db error recovery here,
461  * use the transaction mechanism as previously mentioned */
462  if (rule_dup != NULL)
463  free(rule_dup);
464  return rc;
465 }
