273 static void _blk_free(struct bpf_state *state, struct bpf_blk *blk)
274 {
275  int iter;
276  struct bpf_blk *b_iter;
277  struct bpf_instr *i_iter;
278 
279  if (blk == NULL)
280  return;
281 
282  /* remove this block from the hash table */
283  _hsh_remove(state, blk->hash);
284 
285  /* run through the block freeing TGT_PTR_{BLK,HSH} jump targets */
286  for (iter = 0; iter < blk->blk_cnt; iter++) {
287  i_iter = &blk->blks[iter];
288  switch (i_iter->jt.type) {
289  case TGT_PTR_BLK:
290  _blk_free(state, i_iter->jt.tgt.blk);

292  case TGT_PTR_HSH:
293  b_iter = _hsh_find(state, i_iter->jt.tgt.hash);
294  _blk_free(state, b_iter);
295  break;
296  default:
297  /* do nothing */
298  break;
299  }
300  switch (i_iter->jf.type) {
301  case TGT_PTR_BLK:
302  _blk_free(state, i_iter->jf.tgt.blk);
303  break;
304  case TGT_PTR_HSH:
305  b_iter = _hsh_find(state, i_iter->jf.tgt.hash);
306  _blk_free(state, b_iter);
307  break;
308  default:
309  /* do nothing */
310  break;
311  }
312  }

442 static int _bpf_append_blk(struct bpf_program *prg, const struct bpf_blk *blk)
443 {
444  int rc;
445  bpf_instr_raw *i_new;
446  bpf_instr_raw *i_iter;
447  unsigned int old_cnt = prg->blk_cnt;
448  unsigned int iter;
449  size_t old_size, new_size;
450 
451  /* (re)allocate the program memory */
452  old_size = BPF_PGM_SIZE(prg);
453  prg->blk_cnt += blk->blk_cnt;
454  new_size = BPF_PGM_SIZE(prg);
455  i_new = zrealloc(prg->blks, old_size, new_size);
456  if (i_new == NULL) {
457  rc = -ENOMEM;
458  goto bpf_append_blk_failure;
459  }
460  prg->blks = i_new;
461 
462  /* transfer and translate the blocks to raw instructions */
463  for (iter = 0; iter < blk->blk_cnt; iter++) {
464  i_iter = &(prg->blks[old_cnt + iter]);
465 
466  i_iter->code = blk->blks[iter].op;
467  switch (blk->blks[iter].jt.type) {
468  case TGT_NONE:
469  i_iter->jt = 0;
470  break;
471  case TGT_IMM:
472  /* jump to the value specified */
473  i_iter->jt = blk->blks[iter].jt.tgt.imm_j;
474  break;
475  default:
476  /* fatal error - we should never get here */
477  rc = -EFAULT;
478  goto bpf_append_blk_failure;
479  }
480  switch (blk->blks[iter].jf.type) {
481  case TGT_NONE:
482  i_iter->jf = 0;
483  break;
484  case TGT_IMM:
485  /* jump to the value specified */
486  i_iter->jf = blk->blks[iter].jf.tgt.imm_j;
487  break;
488  default:
489  /* fatal error - we should never get here */
490  rc = -EFAULT;
491  goto bpf_append_blk_failure;
492  }
493  switch (blk->blks[iter].k.type) {
494  case TGT_K:
495  i_iter->k = blk->blks[iter].k.tgt.imm_k;
496  break;
497  default:
498  /* fatal error - we should never get here */
499  rc = -EFAULT;
500  goto bpf_append_blk_failure;
501  }
502  }
503 
504  return prg->blk_cnt;
505 
506 bpf_append_blk_failure:
507  prg->blk_cnt = 0;
508  free(prg->blks);
509  return rc;
510 }

571 static int _hsh_add(struct bpf_state *state, struct bpf_blk **blk_p,
572  unsigned int found)
573 {
574  uint64_t h_val, h_val_tmp[3];
575  struct bpf_hash_bkt *h_new, *h_iter, *h_prev = NULL;
576  struct bpf_blk *blk = *blk_p;
577  struct bpf_blk *b_iter;
578 
579  if (blk->flag_hash)
580  return 0;
581 
582  h_new = zmalloc(sizeof(*h_new));
583  if (h_new == NULL)
584  return -ENOMEM;
585 
586  /* generate the hash */
587  h_val_tmp[0] = hash(blk->blks, _BLK_MSZE(blk));
588  h_val_tmp[1] = hash(&blk->acc_start, sizeof(blk->acc_start));
589  h_val_tmp[2] = hash(&blk->acc_end, sizeof(blk->acc_end));
590  h_val = hash(h_val_tmp, sizeof(h_val_tmp));
591  blk->hash = h_val;
592  blk->flag_hash = true;
593  blk->node = NULL;
594  h_new->blk = blk;
595  h_new->found = (found ? 1 : 0);
596 
597  /* insert the block into the hash table */
598 hsh_add_restart:
599  h_iter = state->htbl[h_val & _BPF_HASH_MASK];
600  if (h_iter != NULL) {
601  do {
602  if ((h_iter->blk->hash == h_val) &&
603  (_BLK_MSZE(h_iter->blk) == _BLK_MSZE(blk)) &&
604  (memcmp(h_iter->blk->blks, blk->blks,
605  _BLK_MSZE(blk)) == 0) &&
606  _ACC_CMP_EQ(h_iter->blk->acc_start,
607  blk->acc_start) &&
608  _ACC_CMP_EQ(h_iter->blk->acc_end,
609  blk->acc_end)) {
610  /* duplicate block */
611  free(h_new);
612 
613  /* store the duplicate block */
614  b_iter = h_iter->blk;
615  while (b_iter->hash_nxt != NULL)
616  b_iter = b_iter->hash_nxt;
617  b_iter->hash_nxt = blk;
618 
619  /* in some cases we want to return the
620  * duplicate block */
621  if (found) {
622  blk->flag_dup = true;
623  return 0;
624  }
625 
626  /* update the priority if needed */
627  if (h_iter->blk->priority < blk->priority)
628  h_iter->blk->priority = blk->priority;
629 
630  /* try to save some memory */
631  free(blk->blks);
632  blk->blks = h_iter->blk->blks;
633  blk->flag_unique = false;
634 
635  *blk_p = h_iter->blk;
636  return 0;
637  } else if (h_iter->blk->hash == h_val) {
638  /* hash collision */
639  if ((h_val >> 32) == 0xffffffff) {
640  /* overflow */
641  blk->flag_hash = false;
642  blk->hash = 0;
643  free(h_new);
644  return -EFAULT;
645  }
646  h_val += ((uint64_t)1 << 32);
647  h_new->blk->hash = h_val;
648 
649  /* restart at the beginning of the bucket */
650  goto hsh_add_restart;
651  } else {
652  /* no match, move along */
653  h_prev = h_iter;
654  h_iter = h_iter->next;
655  }
656  } while (h_iter != NULL);
657  h_prev->next = h_new;
658  } else
659  state->htbl[h_val & _BPF_HASH_MASK] = h_new;
660 
661  return 0;
662 }

828  struct bpf_instr instr;
829

831  if (blk == NULL)
832  return NULL;
833  blk->acc_start = *a_state;
834 
835  /* generate the action blocks */
836  if (node->act_t_flg) {
837  b_act = _gen_bpf_action_hsh(state, node->act_t);
838  if (b_act == NULL)
839  goto node_failure;
840  act_t_hash = b_act->hash;
841  }

847  }

850  acc_offset = node->arg_offset;
851  acc_mask = node->mask;
852  if (acc_offset < 0)

855  ((acc_mask & a_state->mask) != acc_mask)) {
856  /* reload the accumulator */
857  a_state->offset = acc_offset;
858  a_state->mask = ARG_MASK_MAX;
859  _BPF_INSTR(instr, _BPF_OP(state->arch, BPF_LD + BPF_ABS),
860  _BPF_JMP_NO, _BPF_JMP_NO,
861  _BPF_K(state->arch, acc_offset));
862  blk = _blk_append(state, blk, &instr);
863  if (blk == NULL)
864  goto node_failure;
865  /* we're not dependent on the accumulator anymore */
866  blk->acc_start = _ACC_STATE_UNDEF;
867  }
868  if (acc_mask != a_state->mask) {
869  /* apply the bitmask */
870  a_state->mask = acc_mask;
871  _BPF_INSTR(instr, _BPF_OP(state->arch, BPF_ALU + BPF_AND),
872  _BPF_JMP_NO, _BPF_JMP_NO,
873  _BPF_K(state->arch, acc_mask));
874  blk = _blk_append(state, blk, &instr);
875  if (blk == NULL)
876  goto node_failure;
877  }
878 
879  /* set the accumulator state at the end of the block */
880  /* NOTE: the accumulator end state is very critical when we are
881  * assembling the final state; we assume that however we leave
882  * this instruction block the accumulator state is represented
883  * by blk->acc_end, it must be kept correct */
884  blk->acc_end = *a_state;
885 
886  /* check the accumulator against the datum */
887  switch (node->op) {
888  case SCMP_CMP_MASKED_EQ:
889  case SCMP_CMP_EQ:
890  _BPF_INSTR(instr, _BPF_OP(state->arch, BPF_JMP + BPF_JEQ),
891  _BPF_JMP_NO, _BPF_JMP_NO,
892  _BPF_K(state->arch, node->datum));
893  break;
894  case SCMP_CMP_GT:
895  _BPF_INSTR(instr, _BPF_OP(state->arch, BPF_JMP + BPF_JGT),
896  _BPF_JMP_NO, _BPF_JMP_NO,
897  _BPF_K(state->arch, node->datum));
898  break;
899  case SCMP_CMP_GE:
900  _BPF_INSTR(instr, _BPF_OP(state->arch, BPF_JMP + BPF_JGE),
901  _BPF_JMP_NO, _BPF_JMP_NO,
902  _BPF_K(state->arch, node->datum));
903  break;
904  case SCMP_CMP_NE:
905  case SCMP_CMP_LT:
906  case SCMP_CMP_LE:
907  default:
908  /* fatal error, we should never get here */
909  goto node_failure;
910  }
911 
912  /* fixup the jump targets */
913  if (node->nxt_t != NULL)
914  instr.jt = _BPF_JMP_DB(node->nxt_t);
915  else if (node->act_t_flg)
916  instr.jt = _BPF_JMP_HSH(act_t_hash);
917  else
918  instr.jt = _BPF_JMP_NXT(0);
919  if (node->nxt_f != NULL)
920  instr.jf = _BPF_JMP_DB(node->nxt_f);
921  else if (node->act_f_flg)
922  instr.jf = _BPF_JMP_HSH(act_f_hash);
923  else
924  instr.jf = _BPF_JMP_NXT(0);
925  blk = _blk_append(state, blk, &instr);
926  if (blk == NULL)
927  goto node_failure;
928 
929  blk->node = node;
930  return blk;
931 
932 node_failure:
933  _blk_free(state, blk);
934  return NULL;
935 }

1051 static struct bpf_blk *_gen_bpf_chain(struct bpf_state *state,
1052  const struct db_sys_list *sys,
1053  const struct db_arg_chain_tree *chain,
1054  const struct bpf_jump *nxt_jump,
1055  struct acc_state *a_state)
1056 {

1058  struct bpf_blk *b_prev, *b_next, *b_iter;
1059  struct bpf_instr *i_iter;
1060  const struct db_arg_chain_tree *c_iter;
1061  unsigned int iter;
1062  struct bpf_jump nxt_jump_tmp;
1063  struct acc_state acc = *a_state;
1064 
1065  if (chain == NULL) {
1066  b_head = _gen_bpf_action(state, NULL, sys->action);
1067  if (b_head == NULL)
1068  goto chain_failure;
1069  b_tail = b_head;
1070  } else {
1071  /* find the starting node of the level */
1072  c_iter = chain;
1073  while (c_iter->lvl_prv != NULL)
1074  c_iter = c_iter->lvl_prv;
1075 
1076  /* build all of the blocks for this level */
1077  do {
1078  b_iter = _gen_bpf_node(state, c_iter, &acc);
1079  if (b_iter == NULL)
1080  goto chain_failure;
1081  if (b_head != NULL) {
1082  b_iter->lvl_prv = b_tail;
1083  b_tail->lvl_nxt = b_iter;
1084  b_tail = b_iter;
1085  } else {
1086  b_head = b_iter;
1087  b_tail = b_iter;
1088  }
1089  c_iter = c_iter->lvl_nxt;
1090  } while (c_iter != NULL);
1091 
1092  /* resolve the TGT_NXT jumps */
1093  b_iter = b_head;
1094  do {
1095  b_next = b_iter->lvl_nxt;
1096  for (iter = 0; iter < b_iter->blk_cnt; iter++) {
1097  i_iter = &b_iter->blks[iter];
1098  if (i_iter->jt.type == TGT_NXT) {
1099  if (i_iter->jt.tgt.nxt != 0)
1100  goto chain_failure;
1101  if (b_next == NULL)
1102  i_iter->jt = *nxt_jump;
1103  else
1104  i_iter->jt =
1105  _BPF_JMP_BLK(b_next);
1106  }
1107  if (i_iter->jf.type == TGT_NXT) {
1108  if (i_iter->jf.tgt.nxt != 0)
1109  goto chain_failure;
1110  if (b_next == NULL)
1111  i_iter->jf = *nxt_jump;
1112  else
1113  i_iter->jf =
1114  _BPF_JMP_BLK(b_next);
1115  }
1116  }
1117  b_iter = b_next;
1118  } while (b_iter != NULL);
1119  }
1120 
1121  /* resolve all of the blocks */
1122  memset(&nxt_jump_tmp, 0, sizeof(nxt_jump_tmp));
1123  b_iter = b_tail;
1124  do {
1125  /* b_iter may change after resolving, so save the linkage */
1126  b_prev = b_iter->lvl_prv;
1127  b_next = b_iter->lvl_nxt;
1128 
1129  nxt_jump_tmp = _BPF_JMP_BLK(b_next);
1130  b_iter = _gen_bpf_chain_lvl_res(state, sys, b_iter,
1131  (b_next == NULL ?
1132  nxt_jump :
1133  &nxt_jump_tmp));
1134  if (b_iter == NULL)
1135  goto chain_failure;
1136 
1137  /* restore the block linkage on this level */
1138  if (b_prev != NULL)
1139  b_prev->lvl_nxt = b_iter;
1140  b_iter->lvl_prv = b_prev;
1141  b_iter->lvl_nxt = b_next;
1142  if (b_next != NULL)
1143  b_next->lvl_prv = b_iter;
1144  if (b_iter->lvl_prv == NULL)
1145  b_head = b_iter;
1146 
1147  b_iter = b_prev;
1148  } while (b_iter != NULL);
1149 
1150  return b_head;
1151 
1152 chain_failure:
1153  while (b_head != NULL) {
1154  b_iter = b_head;
1155  b_head = b_iter->lvl_nxt;
1156  _blk_free(state, b_iter);
1157  }
1158  return NULL;
1159 }

1414 static int _gen_bpf_bintree(struct bpf_state *state, uint64_t *bintree_hashes,
1415  unsigned int *bintree_syscalls,
1416  unsigned int bintree_levels,
1417  unsigned int total_cnt, unsigned int cur_syscall,
1418  unsigned int *blks_added)
1419 {

1424

1426  level = SYSCALLS_PER_NODE << i;
1427

1438  * tree and the jump-to location is
1439  * not valid. skip this if-else

1444  _BPF_OP(state->arch, BPF_JMP + BPF_JGT),
1445  _BPF_JMP_NO,
1446  _BPF_JMP_NO,
1447  _BPF_K(state->arch, bintree_syscalls[j]));

1449  state->b_new->hash : b_bintree->hash);
1450  instr.jf = _BPF_JMP_HSH(bintree_hashes[j]);

1457  }
1458 
1459  if (b_bintree != NULL)

1461  * save it off so the next binary tree
1462  * if can "else" to it.
1463  */
1464  bintree_hashes[j] = b_bintree->hash;
1465  break;
1466  }
1467  }
1468 
1469 out:
1470  return rc;

1488  const struct db_sys_list *sys,
1489  uint64_t nxt_hash,
1490  bool acc_reset)

1522  blk_s->acc_end = _ACC_STATE_OFFSET(_BPF_OFFSET_SYSCALL);
